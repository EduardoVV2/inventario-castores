CREATE TABLE Avisos (

    IdAvisos BIGINT PRIMARY KEY IDENTITY, -- Identificador único auto-incremental

    Titulo VARCHAR(255) NOT NULL,    -- Título del aviso

    Contenido TEXT NOT NULL,          -- Contenido del aviso

    MedioEnvio INT NOT NULL, -- Medio por el cual se enviará (correo, SMS, etc.)

    FechaEvento DATE NOT NULL,                 -- Fecha del evento relacionado con el aviso

    FechaRegistro DATETIME DEFAULT GETDATE(), -- Fecha en que se registró el aviso (por defecto la actual)

    FechaEnvio DATETIME,              -- Fecha en que se enviará el aviso

    -- CorreoDestinatario NVARCHAR(255), Correo del destinatario

    IdDocumento INT FOREIGN KEY (IdDocumento) REFERENCES Documentos(IdDocumento),           -- Archivo relacionado al aviso (almacenado como binario)

    Activo BIT DEFAULT 1              -- Indica si el aviso está activo (1 = Activo, 0 = Inactivo)

);
